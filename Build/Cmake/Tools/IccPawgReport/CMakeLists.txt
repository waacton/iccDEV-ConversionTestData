SET( SRC_PATH ../../../.. )
SET( SOURCES
  ${SRC_PATH}/Tools/CmdLine/IccPawgReport/IccPawgReport.cpp
  ${SRC_PATH}/Tools/CmdLine/IccPawgReport/PawgReport.cpp
)
SET( TARGET_NAME iccPawgReport )

ADD_EXECUTABLE( ${TARGET_NAME} ${SOURCES} )
TARGET_LINK_LIBRARIES( ${TARGET_NAME} ${TARGET_LIB_ICCPROFLIB} )
IF(WIN32 AND ENABLE_SHARED_LIBS)
  ADD_CUSTOM_COMMAND(TARGET ${TARGET_NAME} POST_BUILD

    COMMAND_EXPAND_LISTS)
ENDIF()
IF(ENABLE_INSTALL_RIM)
  INSTALL (TARGETS ${TARGET_NAME} DESTINATION ${CMAKE_INSTALL_BINDIR})
ENDIF(ENABLE_INSTALL_RIM)
